#include <iostream>
using namespace std;
class shape { //abstract class
public:
virtual void area()=0;
void display() {
    cout<<"this is a shape."<<endl; }
};
class circle:public shape {
    public:
    void area() {
        cout<<"area of a circle=22/7*r*r"<<endl; }
    };
    class square:public shape{
        public:
        void area() {
            cout<<"Area of square=side*side"<<endl;
        }
    };
    int main() {
        circle c;
        square s;
        c.display();
        c.area();
        s.area();
    }
